\type[\Var{}]
 {{\begin{array}{l}
    \ensuremath{\gamma} : \Context{n} \\
    \ensuremath{\Gamma} : \Usages{\ensuremath{\gamma}} \\ \\
  \end{array}}
  \and k : \Var{n}
  \and {\begin{array}{l}
    \ensuremath{\sigma} : \Type{}    \\
    \ensuremath{\Delta} : \Usages{\ensuremath{\gamma}} \\ \\
  \end{array}}
}{\ensuremath{\Gamma} \ensuremath{\vdash}_v k \ensuremath{\in} \ensuremath{\sigma} \andalso{} \ensuremath{\Delta} : \Set{}
}
\and \type[\Inferable{}]
 {{\begin{array}{l}
    \ensuremath{\gamma} : \Context{n} \\
    \ensuremath{\Gamma} : \Usages{\ensuremath{\gamma}} \\ \\
  \end{array}}
  \and t : \Inferable{n}
  \and {\begin{array}{l}
    \ensuremath{\sigma} : \Type{}    \\
    \ensuremath{\Delta} : \Usages{\ensuremath{\gamma}} \\ \\
  \end{array}}
}{\ensuremath{\Gamma} \ensuremath{\vdash} t \ensuremath{\in} \ensuremath{\sigma} \andalso{} \ensuremath{\Delta} : \Set{}
}
\and \type[\Checkable{}]
 {{\begin{array}{l}
    \ensuremath{\gamma} : \Context{n} \\
    \ensuremath{\Gamma} : \Usages{\ensuremath{\gamma}} \\
    \ensuremath{\sigma} : \Type{} \\ \\
   \end{array}}
  \and t : \Checkable{n}
  \and \ensuremath{\Delta} : \Usages{\ensuremath{\gamma}}
}{\ensuremath{\Gamma} \ensuremath{\vdash} \ensuremath{\sigma}  \ensuremath{\ni} t \andalso{} \ensuremath{\Delta} : \Set{}
}
\and \type[\Pattern{}]
{ \ensuremath{\sigma} : \Type{}
  \and p : \Pattern{n}
  \and \ensuremath{\gamma} : \Context{n}
}{\ensuremath{\sigma} \ensuremath{\ni} p \leadsto{} \ensuremath{\gamma} : \Set{}
}
\end{mathpar}
