\and \inferrule
 {Γ ⊢ t ∈ σ ⊕ τ ⊠ Δ \and
  {\begin{array}{l}

    Δ ∙ \fresh{τ} ⊢ ν ∋ r ⊠ Θ ∙ \stale{τ}
  \end{array}}
}{Γ ⊢ \cas{t}{ν}{l}{r} ∈ ν ⊠ Θ
}
\and \inferrule

}{Γ ⊢ \cut{t}{σ} ∈ σ ⊠ Δ
