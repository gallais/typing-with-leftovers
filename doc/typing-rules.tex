\type[\Var{}]
 {{\begin{array}{l}
    γ : \Context{n} \\
    Γ : \Usages{γ} \\ \\
  \end{array}}
  \and k : \Var{n}
  \and {\begin{array}{l}
    σ : \Type{}    \\
    Δ : \Usages{γ} \\ \\
  \end{array}}
}{Γ ⊢_v k ∈ σ \andalso{} Δ : \Set{}
}
\and \type[\Inferable{}]
 {{\begin{array}{l}
    γ : \Context{n} \\
    Γ : \Usages{γ} \\ \\
  \end{array}}
  \and t : \Inferable{n}
  \and {\begin{array}{l}
    σ : \Type{}    \\
    Δ : \Usages{γ} \\ \\
  \end{array}}
}{Γ ⊢ t ∈ σ \andalso{} Δ : \Set{}
}
\and \type[\Checkable{}]
 {{\begin{array}{l}
    γ : \Context{n} \\
    Γ : \Usages{γ} \\
    σ : \Type{} \\ \\
   \end{array}}
  \and t : \Checkable{n}
  \and Δ : \Usages{γ}
}{Γ ⊢ σ  ∋ t \andalso{} Δ : \Set{}
}
\and \type[\Pattern{}]
{ σ : \Type{}
  \and p : \Pattern{n}
  \and γ : \Context{n}
}{σ ∋ p \leadsto{} γ : \Set{}
}
\end{mathpar}
